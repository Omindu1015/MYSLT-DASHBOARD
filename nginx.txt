server {
    listen 443 ssl default_server;
    server_name yourdomain.com; # Replace with your actual domain

    # SSL Certificate Paths (Converted from .pfx)
    ssl_certificate     /etc/nginx/ssl/myslt_fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/myslt_privkey.pem;

    # Recommended SSL Settings
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    # Global security hardening
    server_tokens off; # Hides Nginx version
    
    # Global header for verification
    add_header X-Nginx-Config "MySLT-Dashboard-v1";
    
    # FIXED: Security Headers (Vulnerability Mitigation)
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header Content-Security-Policy "default-src 'self' https://api.emailjs.com; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline'; img-src 'self' data:; connect-src 'self' https://api.emailjs.com; frame-ancestors 'self';" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    # 1. Frontend: React Application
    location / {
        root /var/www/MYSLT-DASHBOARD/client/dist;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    # FIXED: Cache Control for Auth-Sensitive Pages
    location ~* /(login|admin) {
        root /var/www/MYSLT-DASHBOARD/client/dist;
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate" always;
        add_header Pragma "no-cache" always;
        try_files $uri $uri/ /index.html;
    }

    # 2. Backend: Node.js API
    location /api/ {
        proxy_pass http://127.0.0.1:5001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Hide backend technology info
        proxy_hide_header X-Powered-By;
    }

    # Custom JSON error handler for internal errors
    error_page 502 503 504 /json_error;
    location = /json_error {
        internal;
        add_header Content-Type application/json;
        return 502 '{"error": "Backend Gateway Error", "path": "$request_uri"}';
    }
}
